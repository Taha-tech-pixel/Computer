Language Overview

VB.NET (Visual Basic .NET) is a modern, object-oriented programming language developed by Microsoft as part of the .NET Framework. It evolved from the classic Visual Basic language and provides a simple, readable syntax while offering the full power of the .NET platform. VB.NET is widely used for Windows applications, web development, and enterprise software.

File Operations

Reading Files

Imports System.IO

Module FileOperations
    Sub ReadFile()
        Try
            ' Read all text
            Dim content As String = File.ReadAllText("data.txt")
            Console.WriteLine(content)
            
            ' Read lines
            Dim lines() As String = File.ReadAllLines("data.txt")
            For Each line In lines
                Console.WriteLine(line)
            Next
            
            ' Stream reader
            Using reader As New StreamReader("data.txt")
                Dim line As String
                While (line = reader.ReadLine()) IsNot Nothing
                    Console.WriteLine(line)
                End While
            End Using
            
        Catch ex As FileNotFoundException
            Console.WriteLine("File not found")
        Catch ex As Exception
            Console.WriteLine($"Error reading file: {ex.Message}")
        End Try
    End Sub
End Module

Public Interface IRepository(Of T)
    Function GetByIdAsync(id As Integer) As Task(Of T)
    Function GetAllAsync() As Task(Of IEnumerable(Of T))
    Function AddAsync(entity As T) As Task(Of T)
    Function UpdateAsync(entity As T) As Task(Of T)
    Function DeleteAsync(id As Integer) As Task(Of Boolean)
End Interface

Public Class Repository(Of T As Class)
    Implements IRepository(Of T)
    
    Private ReadOnly _context As DbContext
    Private ReadOnly _dbSet As DbSet(Of T)
    
    Public Sub New(context As DbContext)
        _context = context
        _dbSet = context.Set(Of T)()
    End Sub
    
    Public Async Function GetByIdAsync(id As Integer) As Task(Of T) Implements IRepository(Of T).GetByIdAsync
        Return Await _dbSet.FindAsync(id)
    End Function
    
    Public Async Function GetAllAsync() As Task(Of IEnumerable(Of T)) Implements IRepository(Of T).GetAllAsync
        Return Await _dbSet.ToListAsync()
    End Function
    
    Public Async Function AddAsync(entity As T) As Task(Of T) Implements IRepository(Of T).AddAsync
        _dbSet.Add(entity)
        Await _context.SaveChangesAsync()
        Return entity
    End Function
End Class

' LINQ and Lambda expressions
Public Class DataProcessor
    Public Shared Function ProcessUserData(users As IEnumerable(Of User)) As IEnumerable(Of UserSummary)
        Return users _
            .Where(Function(u) u.IsActive AndAlso u.Age >= 18) _
            .GroupBy(Function(u) u.Department) _
            .Select(Function(g) New UserSummary With {
                .Department = g.Key,
                .UserCount = g.Count(),
                .AverageAge = g.Average(Function(u) u.Age),
                .TotalSalary = g.Sum(Function(u) u.Salary)
            }) _
            .OrderByDescending(Function(s) s.TotalSalary)
    End Function
    
    ' Advanced LINQ operations
    Public Shared Function ComplexQuery(orders As IEnumerable(Of Order)) As Object
        Return From o In orders
               Where o.OrderDate >= DateTime.Today.AddMonths(-3)
               Group o By New With {o.CustomerId, CustomerName = o.Customer.Name} Into g = Group
               Let totalAmount = g.Sum(Function(x) x.Amount)
               Let orderCount = g.Count()
               Where totalAmount > 1000
               Order By totalAmount Descending
               Select New With {
                   .CustomerId = g.Key.CustomerId,
                   .CustomerName = g.Key.CustomerName,
                   .TotalAmount = totalAmount,
                   .OrderCount = orderCount,
                   .AverageOrderValue = totalAmount / orderCount
               }
    End Function
End Class

' Async/Await patterns
Public Class WebApiService
    Private ReadOnly _httpClient As HttpClient

    Public Async Function GetUserDataAsync(userId As Integer) As Task(Of UserDto)
        Try
            Dim response = Await _httpClient.GetAsync($"api/users/{userId}")
            response.EnsureSuccessStatusCode()
            
            Dim jsonString = Await response.Content.ReadAsStringAsync()

            Throw New ApplicationException($"Failed to retrieve user {userId}", ex)
        End Try
    End Function

    Public Async Function GetMultipleUsersAsync(userIds As Integer()) As Task(Of IEnumerable(Of UserDto))
        Dim tasks = userIds.Select(Function(id) GetUserDataAsync(id))
        Return Await Task.WhenAll(tasks)
    End Function

        RaiseEvent PropertyChanged(Me, New PropertyChangedEventArgs(propertyName))

        If EqualityComparer(Of T).Default.Equals(field, value) Then

        End If
        
        field = value
        OnPropertyChanged(propertyName)
        Return True
    End Function

Public Class UsersController

    Public Sub New(userService As IUserService, logger As ILogger(Of UsersController))
        _userService = userService
        _logger = logger
    End Sub
    
    
    Public Async Function GetUsers( pageNumber As Integer = 1,  pageSize As Integer = 10) As Task(Of ActionResult(Of PagedResult(Of UserDto)))
        Try
            Dim result = Await _userService.GetUsersAsync(pageNumber, pageSize)
            Return Ok(result)
        Catch ex As Exception
            _logger.LogError(ex, "Error retrieving users")
            Return StatusCode(500, "Internal server error")
        End Try
    End Function
    
    
    Public Async Function GetUser(id As Integer) As Task(Of ActionResult(Of UserDto))
        Dim user = Await _userService.GetUserByIdAsync(id)
        If user Is Nothing Then
            Return NotFound()
        End If
        Return Ok(user)
    End Function
    
    
    
    Public Async Function CreateUser( createUserDto As CreateUserDto) As Task(Of ActionResult(Of UserDto))
        If Not ModelState.IsValid Then
            Return BadRequest(ModelState)
        End If
        
        Try
            Dim user = Await _userService.CreateUserAsync(createUserDto)
            Return CreatedAtAction(NameOf(GetUser), New With {.id = user.Id}, user)
        Catch ex As ValidationException
            Return BadRequest(ex.Message)
        End Try
    End Function
    
    
    
    Public Async Function UpdateUser(id As Integer,  updateUserDto As UpdateUserDto) As Task(Of ActionResult)
        If id <> updateUserDto.Id Then
            Return BadRequest("ID mismatch")
        End If
        
        Try
            Await _userService.UpdateUserAsync(updateUserDto)
            Return NoContent()
        Catch ex As NotFoundException
            Return NotFound()
        End Try
    End Function
End Class

' Middleware for request/response logging
Public Class RequestLoggingMiddleware
    Private ReadOnly _next As RequestDelegate
    Private ReadOnly _logger As ILogger(Of RequestLoggingMiddleware)
    
    Public Sub New(next As RequestDelegate, logger As ILogger(Of RequestLoggingMiddleware))
        _next = next
        _logger = logger
    End Sub
    
    Public Async Function InvokeAsync(context As HttpContext) As Task
        Dim stopwatch = Stopwatch.StartNew()
        
        _logger.LogInformation($"Request: {context.Request.Method} {context.Request.Path}")
        
        Await _next(context)
        
        stopwatch.Stop()
        _logger.LogInformation($"Response: {context.Response.StatusCode} in {stopwatch.ElapsedMilliseconds}ms")
    End Function
End Class

Public Class Startup
    Public Sub ConfigureServices(services As IServiceCollection)
        services.AddControllers()
        services.AddDbContext(Of AppDbContext)(Function(options) options.UseSqlServer(connectionString))
        
        ' Register services

Public Module StringExtensions
    
    Public Function ToTitleCase(source As String) As String
        If String.IsNullOrEmpty(source) Then Return source
        
        Return Threading.Thread.CurrentThread.CurrentCulture.TextInfo.ToTitleCase(source.ToLower())
    End Function
    
    
    Public Function IsValidEmail(source As String) As Boolean
        Const pattern = "^[\w-\.]+@([\w-]+\.)+[\w-]{2,4}$"
        Return Regex.IsMatch(source, pattern)
    End Function
    
    
    Public Function ToSlug(source As String) As String
        Return Regex.Replace(source.ToLower().Trim(), "[^a-z0-9\s-]", "") _
                   .Replace(" ", "-") _
                   .Replace("--", "-")
    End Function

Public Class DataAnalyzer
    Public Shared Function AnalyzeData(data As IEnumerable(Of Object)) As Object
        Dim results = From item In data
                      Let itemType = item.GetType()
                      Group item By itemType Into g = Group
                      Select New With {
                          .Type = itemType.Name,
                          .Count = g.Count(),
                          .Items = g.ToList()
                      }
        
        Return New With {
            .TotalItems = data.Count(),
            .TypeBreakdown = results.ToList(),
            .AnalyzedAt = DateTime.Now
        }
    End Function
End Class

    Public Shared Function ParseUserXml(xml As XDocument) As IEnumerable(Of User)
        Return From userElement In xml.Root.Elements("user")
               Select New User With {
                   .Id = CInt(userElement.Attribute("id").Value),
                   .Name = userElement.Element("name").Value,
                   .Email = userElement.Element("email").Value,
                   .Department = userElement.Element("department").Value,
                   .IsActive = CBool(userElement.Element("isActive").Value)
               }
    End Function
End Class

' Advanced exception handling
Public Class CustomException
    Inherits Exception
    
    Public Property ErrorCode As String
    Public Property UserMessage As String
    
    Public Sub New(errorCode As String, userMessage As String, Optional innerException As Exception = Nothing)
        MyBase.New($"{errorCode}: {userMessage}", innerException)
        Me.ErrorCode = errorCode
        Me.UserMessage = userMessage
    End Sub
End Class

Public Class ExceptionHandler
    Public Shared Function HandleException(Of T)(action As Func(Of T)) As Result(Of T)
        Try
            Dim result = action()
            Return Result(Of T).Success(result)
        Catch ex As CustomException
            Return Result(Of T).Failure(ex.ErrorCode, ex.UserMessage)
        Catch ex As Exception
            Return Result(Of T).Failure("UNKNOWN_ERROR", "An unexpected error occurred")
        End Try
    End Function
End Class

Public Class Result(Of T)
    Public Property IsSuccess As Boolean
    Public Property Value As T
    Public Property ErrorCode As String
    Public Property ErrorMessage As String
    
    Public Shared Function Success(value As T) As Result(Of T)
        Return New Result(Of T) With {
            .IsSuccess = True,
            .Value = value
        }
    End Function
    
    Public Shared Function Failure(errorCode As String, errorMessage As String) As Result(Of T)
        Return New Result(Of T) With {
            .IsSuccess = False,
            .ErrorCode = errorCode,
            .ErrorMessage = errorMessage
        }
    End Function
End Class

' Pattern matching (VB.NET 16+)
Public Class PatternMatcher
    Public Shared Function ProcessValue(value As Object) As String
        Return value switch {
            String s When s.Length > 10 => $"Long string: {s.Substring(0, 10)}...",
            String s => $"Short string: {s}",
            Integer i When i > 100 => $"Large number: {i}",
            Integer i => $"Small number: {i}",
            Boolean b => $"Boolean: {b}",
            Nothing => "Null value",
            _ => $"Unknown type: {value.GetType().Name}"
        }
    End Function
End Class
