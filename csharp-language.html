Language Overview

C# (pronounced "C sharp") is a modern, object-oriented programming language developed by Microsoft as part of the .NET initiative. It combines the power and flexibility of C++ with the simplicity of Visual Basic, making it ideal for building Windows applications, web services, and enterprise software.

using System;

// Basic class
public class Person
{
    // Fields
    private string name;
    private int age;
    
    // Properties
    public string Name
    {
        get { return name; }
        set { name = value; }
    }
    
    public int Age
    {
        get { return age; }
        set 
        { 
            if (value >= 0)
                age = value; 
        }
    }
    
    // Auto-implemented properties
    public string Email { get; set; }
    
    // Constructor
    public Person(string name, int age)
    {
        this.name = name;
        this.age = age;
    }
    
    // Method
    public virtual void Display()
    {
        Console.WriteLine($"Name: {Name}, Age: {Age}");
    }
    
    // Static method
    public static Person CreateAdult(string name)
    {
        return new Person(name, 18);
    }
}

// Inheritance
public class Student : Person
{
    public string StudentId { get; set; }
    
    public Student(string name, int age, string studentId)

        StudentId = studentId;
    }

    }

Interfaces and Abstract Classes

using System;
using System.Collections.Generic;

// Interface
public interface IShape
{
    double CalculateArea();
    double CalculatePerimeter();
    void Display();
}

// Abstract class
public abstract class Shape : IShape
{
    public string Color { get; set; }
    
    public abstract double CalculateArea();
    public abstract double CalculatePerimeter();
    
    public virtual void Display()
    {
        Console.WriteLine($"Shape Color: {Color}");
    }
}

// Concrete implementation
public class Circle : Shape
{
    public double Radius { get; set; }
    
    public Circle(double radius, string color)
    {
        Radius = radius;
        Color = color;
    }
    
    public override double CalculateArea()
    {
        return Math.PI * Radius * Radius;
    }
    
    public override double CalculatePerimeter()
    {
        return 2 * Math.PI * Radius;
    }
    
    public override void Display()
    {
        base.Display();
        Console.WriteLine($"Circle - Radius: {Radius}");
    }
}

// Generic interface
public interface IRepository<T>
{
    T GetById(int id);
    IEnumerable<T> GetAll();
    void Add(T entity);
    void Update(T entity);
    void Delete(int id);
}

Exception Handling

using System;

class ExceptionHandling
{
    static void Main()
    {
        try
        {
            // Risky operation
            int result = Divide(10, 0);
            Console.WriteLine($"Result: {result}");
        }
        catch (DivideByZeroException ex)
        {
            Console.WriteLine($"Division by zero error: {ex.Message}");
        }
        catch (Exception ex)
        {
            Console.WriteLine($"Unexpected error: {ex.Message}");
        }
        finally
        {
            Console.WriteLine("Cleanup code executed");
        }
        
        // Custom exception
        try
        {
            ValidateAge(-5);
        }
        catch (InvalidAgeException ex)
        {
            Console.WriteLine($"Validation error: {ex.Message}");
        }
    }
    
    static int Divide(int a, int b)
    {
        if (b == 0)
            throw new DivideByZeroException("Cannot divide by zero");
        return a / b;
    }
    
    static void ValidateAge(int age)
    {
        if (age < 0)
            throw new InvalidAgeException("Age cannot be negative");
    }
}

// Custom exception
public class InvalidAgeException : Exception
{
    public InvalidAgeException(string message) : base(message) { }
}

Async Programming

using System;
using System.Net.Http;
using System.Threading.Tasks;

class AsyncExample
{
    static async Task Main()
    {
        Console.WriteLine("Starting async operations...");
        
        // Sequential execution
        var result1 = await FetchDataAsync("https://api.example.com/data1");
        var result2 = await FetchDataAsync("https://api.example.com/data2");
        
        Console.WriteLine($"Sequential results: {result1}, {result2}");
        
        // Parallel execution
        var task1 = FetchDataAsync("https://api.example.com/data1");
        var task2 = FetchDataAsync("https://api.example.com/data2");
        
        await Task.WhenAll(task1, task2);
        
        Console.WriteLine($"Parallel results: {task1.Result}, {task2.Result}");
        
        // Task with cancellation
        using (var cts = new System.Threading.CancellationTokenSource())
        {
            cts.CancelAfter(TimeSpan.FromSeconds(5)); // Cancel after 5 seconds
            
            try
            {
                var result = await FetchDataWithTimeoutAsync("https://api.example.com/slow", cts.Token);
                Console.WriteLine($"Result: {result}");
            }
            catch (TaskCanceledException)
            {
                Console.WriteLine("Operation was cancelled due to timeout");
            }
        }
    }
    
    static async Task<string> FetchDataAsync(string url)
    {
        using (var client = new HttpClient())
        {
            await Task.Delay(1000); // Simulate network delay
            return $"Data from {url}";
        }
    }
    
    static async Task<string> FetchDataWithTimeoutAsync(string url, System.Threading.CancellationToken cancellationToken)
    {
        using (var client = new HttpClient())
        {
            await Task.Delay(10000, cancellationToken); // Simulate long operation
            return $"Data from {url}";
        }
    }
}
