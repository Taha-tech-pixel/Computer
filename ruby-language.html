Language Overview

Ruby is the foundation of the Ruby on Rails web framework and is widely used for web development, automation, and general-purpose programming.

Basic Syntax

Hello World

puts "Hello, World!"

# Simple method definition
def greet(name)
  "Hello, #{name}!"
end

puts greet("Ruby")

Variables and Data Types

# Local variables
name = "Ruby"
age = 25
price = 19.99

# Constants
PI = 3.14159

# Arrays
colors = ["red", "green", "blue"]

# Hashes
person = { name: "John", age: 30 }

puts "Name: #{person[:name]}"

String Interpolation

name = "World"
greeting = "Hello, #{name}!"
puts greeting

# Alternative syntax
puts "Hello, " + name + "!"
puts "Hello, %s!" % name

Control Structures

Conditionals

age = 18

if age >= 18
  puts "Adult"
elsif age >= 13
  puts "Teenager"
else
  puts "Child"
end

# Postfix if
puts "Adult" if age >= 18

# Unless (opposite of if)
puts "Minor" unless age >= 18

Loops and Iterators

# Traditional for loop
for i in 1..5
  puts i
end

# Each iterator (preferred)
(1..5).each { |i| puts i }

# Times iterator
5.times { |i| puts i + 1 }

# While loop
i = 0
while i < 5
  puts i
  i += 1
end

Classes and Objects

Class Definition

class Person
  # Class variable
  @@count = 0
  
  # Constructor
  def initialize(name, age)
    @name = name
    @age = age
    @@count += 1
  end
  
  # Getter methods
  attr_reader :name, :age
  
  # Instance method
  def introduce
    "Hi, I'm #{@name} and I'm #{@age} years old"
  end
  
  # Class method
  def self.count
    @@count
  end
end

person = Person.new("Alice", 25)
puts person.introduce
puts "Total people: #{Person.count}"

Inheritance

class Employee < Person
  def initialize(name, age, salary)
    super(name, age)
    @salary = salary
  end
  
  def work
    "#{@name} is working hard"
  end
end

employee = Employee.new("Bob", 30, 50000)
puts employee.introduce
puts employee.work

Blocks and Procs

Blocks

# Block with do...end
[1, 2, 3, 4, 5].each do |num|
  puts num * 2
end

# Block with curly braces
[1, 2, 3, 4, 5].each { |num| puts num * 2 }

# Custom method with block
def repeat(times)
  times.times { yield }
end

repeat(3) { puts "Hello!" }

Procs and Lambdas

# Proc
double = Proc.new { |x| x * 2 }
puts double.call(5)  # 10

# Lambda
square = lambda { |x| x ** 2 }
puts square.call(4)  # 16

# Method that takes a proc
def apply_operation(numbers, operation)
  numbers.map(&operation)
end

result = apply_operation([1, 2, 3], double)
puts result.inspect  # [2, 4, 6]

Modules and Mixins

Module Definition

module MathUtils
  PI = 3.14159
  
  def self.square(x)
    x * x
  end
  
  def self.circle_area(radius)
    PI * radius * radius
  end
end

puts MathUtils::PI
puts MathUtils.square(5)
puts MathUtils.circle_area(3)

Mixins

module Swimmable
  def swim
    "I can swim!"
  end
end

module Flyable
  def fly
    "I can fly!"
  end
end

class Duck
  include Swimmable
  include Flyable
  
  def quack
    "Quack!"
  end
end

duck = Duck.new
puts duck.quack
puts duck.swim
puts duck.fly

File Operations

Reading Files

# Read entire file
content = File.read("file.txt")
puts content

# Read line by line
File.foreach("file.txt") do |line|
  puts line.chomp
end

# Read with encoding
content = File.read("file.txt", encoding: "UTF-8")

Writing Files

# Write to file
File.write("output.txt", "Hello, Ruby!")

# Append to file
File.open("log.txt", "a") do |file|
  file.puts Time.now
  file.puts "Log entry"
end

Error Handling

Exception Handling

begin
  # Risky code
  result = 10 / 0
rescue ZeroDivisionError => e
  puts "Error: #{e.message}"
rescue => e
  puts "Unexpected error: #{e.message}"
else
  puts "No errors occurred"
ensure
  puts "This always runs"
end

Custom Exceptions

class ValidationError < StandardError
  def initialize(message)
    super(message)
  end
end

def validate_age(age)
  if age < 0
    raise ValidationError, "Age cannot be negative"
  end
  puts "Valid age: #{age}"
end

begin
  validate_age(-5)
rescue ValidationError => e
  puts "Validation failed: #{e.message}"
end

Best Practices

Follow Ruby conventions

Use snake_case for methods and variables, CamelCase for classes

Prefer blocks over loops

Use each, map, select, and other enumerable methods

Use symbols for hash keys

Symbols are more memory efficient than strings

Handle exceptions properly

Use specific exception types and provide meaningful messages
