        {
            Console.WriteLine("Cleanup code executed");

using System;
using System.Net.Http;
using System.Threading.Tasks;

class AsyncExample
{
    static async Task Main()
    {
        Console.WriteLine("Starting async operations...");
        
        // Sequential execution
        var result1 = await FetchDataAsync("https://api.example.com/data1");
        var result2 = await FetchDataAsync("https://api.example.com/data2");
        
        Console.WriteLine($"Sequential results: {result1}, {result2}");
        
        // Parallel execution
        var task1 = FetchDataAsync("https://api.example.com/data1");
        var task2 = FetchDataAsync("https://api.example.com/data2");
        
        await Task.WhenAll(task1, task2);
        
        Console.WriteLine($"Parallel results: {task1.Result}, {task2.Result}");
        
        // Task with cancellation
        using (var cts = new System.Threading.CancellationTokenSource())
        {
            cts.CancelAfter(TimeSpan.FromSeconds(5)); // Cancel after 5 seconds
            
            try
            {
                var result = await FetchDataWithTimeoutAsync("https://api.example.com/slow", cts.Token);
                Console.WriteLine($"Result: {result}");
            }
            catch (TaskCanceledException)
            {
                Console.WriteLine("Operation was cancelled due to timeout");
            }
        }
    }
    
    static async Task<string> FetchDataAsync(string url)
    {
        using (var client = new HttpClient())
        {
            await Task.Delay(1000); // Simulate network delay
            return $"Data from {url}";
        }
    }
    
    static async Task<string> FetchDataWithTimeoutAsync(string url, System.Threading.CancellationToken cancellationToken)
    {
        using (var client = new HttpClient())
        {
            await Task.Delay(10000, cancellationToken); // Simulate long operation
            return $"Data from {url}";
        }
    }
}

class AsyncProgrammingExamples

    // Async factory pattern
    public static async Task<DataProcessor> CreateAsync(string connectionString)
    {
        var processor = new DataProcessor();
        await processor.InitializeAsync(connectionString);
        return processor;
    }

    // Parallel processing with cancellation
    public static async Task<List<Result>> ProcessDataParallelAsync(
        IEnumerable<string> urls, 
        CancellationToken cancellationToken = default)
    {
        var tasks = urls.Select(url => ProcessUrlAsync(url, cancellationToken));
        var results = await Task.WhenAll(tasks);
        return results.ToList();
    }

    // Async stream processing
    public static async IAsyncEnumerable<string> ProcessStreamAsync(
        IAsyncEnumerable<string> stream)
    {
        await foreach (var item in stream)
        {
            var processed = await ProcessItemAsync(item);
            yield return processed;
        }
    }

    // Retry pattern with exponential backoff
    public static async Task<T> RetryWithBackoffAsync<T>(
        Func<Task<T>> operation,
        int maxRetries = 3,
        TimeSpan? initialDelay = null)
    {
        var delay = initialDelay ?? TimeSpan.FromSeconds(1);
        var lastException = default(Exception);

        for (int i = 0; i <= maxRetries; i++)
        {
            try
            {
                return await operation();
            }
            catch (Exception ex) when (i < maxRetries)
            {
                lastException = ex;
                await Task.Delay(delay);
                delay = TimeSpan.FromTicks(delay.Ticks * 2); // Exponential backoff
            }
        }

        throw lastException ?? new InvalidOperationException("Operation failed");
    }

    // Async event handling
    public class AsyncEvent<T>
    {
        private readonly List<Func<T, Task>> handlers = new();

        public void Subscribe(Func<T, Task> handler)
        {
            lock (handlers)
            {
                handlers.Add(handler);
            }
        }

        public async Task RaiseAsync(T args)
        {
            List<Func<T, Task>> currentHandlers;
            lock (handlers)
            {
                currentHandlers = new List<Func<T, Task>>(handlers);
            }

            var tasks = currentHandlers.Select(handler => handler(args));
            await Task.WhenAll(tasks);
        }
    }

    // Async resource management
    public class AsyncResourceManager : IAsyncDisposable
    {
        private bool disposed = false;

        public async ValueTask DisposeAsync()
        {
            if (!disposed)
            {
                await CleanupAsync();
                disposed = true;
            }
        }

        private async Task CleanupAsync()
        {
            // Async cleanup logic
            await Task.Delay(100);
        }
    }

    // Async pipeline processing
    public static async Task<T> PipelineAsync<T>(
        T input,
        params Func<T, Task<T>>[] steps)
    {
        var result = input;
        foreach (var step in steps)
        {
            result = await step(result);
        }

using System;
using System.Reflection;
using System.Dynamic;
using System.Linq;

// Advanced reflection examples
class ReflectionExamples
{
    // Dynamic object creation
    public static T CreateInstance<T>(params object[] args)
    {
        var type = typeof(T);
        return (T)Activator.CreateInstance(type, args);
    }

    // Property mapping with reflection
    public static T MapProperties<T>(object source) where T : new()
    {
        var target = new T();
        var sourceType = source.GetType();
        var targetType = typeof(T);

        var sourceProperties = sourceType.GetProperties();
        var targetProperties = targetType.GetProperties();

        foreach (var sourceProp in sourceProperties)
        {
            var targetProp = targetProperties.FirstOrDefault(p => 
                p.Name == sourceProp.Name && 
                p.PropertyType == sourceProp.PropertyType);

            if (targetProp != null && targetProp.CanWrite)
            {
                var value = sourceProp.GetValue(source);
                targetProp.SetValue(target, value);
            }
        }

        return target;
    }

    // Method invocation with reflection
    public static object InvokeMethod(object instance, string methodName, params object[] args)
    {
        var method = instance.GetType().GetMethod(methodName);
        if (method == null)
            throw new ArgumentException($"Method {methodName} not found");

        return method.Invoke(instance, args);
    }

    // Attribute discovery
    public static IEnumerable<T> GetAttributes<T>(Type type) where T : Attribute
    {
        return type.GetCustomAttributes(typeof(T), true).Cast<T>();
    }

    // Dynamic object implementation
    public class DynamicDictionary : DynamicObject
    {
        private readonly Dictionary<string, object> dictionary = new();

        public override bool TryGetMember(GetMemberBinder binder, out object result)
        {
            return dictionary.TryGetValue(binder.Name, out result);
        }

        public override bool TrySetMember(SetMemberBinder binder, object value)
        {
            dictionary[binder.Name] = value;
            return true;
        }

        public override bool TryInvokeMember(InvokeMemberBinder binder, object[] args, out object result)
        {
            if (binder.Name == "Count")
            {
                result = dictionary.Count;
                return true;
            }

            result = null;
            return false;
        }
    }

    // Expression trees for dynamic queries
    public static IQueryable<T> DynamicWhere<T>(
        IQueryable<T> source, 
        string propertyName, 
        object value)
    {
        var parameter = Expression.Parameter(typeof(T), "x");
        var property = Expression.Property(parameter, propertyName);
        var constant = Expression.Constant(value);
        var comparison = Expression.Equal(property, constant);
        var lambda = Expression.Lambda<Func<T, bool>>(comparison, parameter);

        return source.Where(lambda);
    }

    // Code generation with reflection
    public static Delegate CreateDelegate(Type delegateType, object target, string methodName)
    {
        var method = target.GetType().GetMethod(methodName);
        return Delegate.CreateDelegate(delegateType, target, method);
    }
}

Memory Management & Performance

using System;
using System.Buffers;
using System.Collections.Generic;
using System.Runtime.CompilerServices;
using System.Threading;

// Memory management and performance optimization
class PerformanceExamples
{
    // Object pooling
    public class ObjectPool<T> where T : class, new()
    {
        private readonly Stack<T> pool = new();
        private readonly object lockObject = new();

        public T Rent()
        {
            lock (lockObject)
            {
                return pool.Count > 0 ? pool.Pop() : new T();
            }
        }

        public void Return(T item)
        {
            lock (lockObject)
            {
                pool.Push(item);
            }
        }
    }

    // Array pool usage
    public static void ProcessWithArrayPool(byte[] data)
    {
        var buffer = ArrayPool<byte>.Shared.Rent(data.Length);
        try
        {
            Array.Copy(data, buffer, data.Length);
            ProcessBuffer(buffer, data.Length);
        }
        finally
        {
            ArrayPool<byte>.Shared.Return(buffer);
        }
    }

    // Memory-efficient string building
    public static string BuildLargeString(IEnumerable<string> parts)
    {
        var builder = new System.Text.StringBuilder();
        foreach (var part in parts)
        {
            builder.Append(part);
        }
        return builder.ToString();
    }

    // Struct for performance
    [StructLayout(LayoutKind.Sequential)]
    public struct Point3D
    {
        public float X, Y, Z;

        public Point3D(float x, float y, float z)
        {
            X = x; Y = y; Z = z;
        }

        public float DistanceTo(Point3D other)
        {
            var dx = X - other.X;
            var dy = Y - other.Y;
            var dz = Z - other.Z;
            return (float)Math.Sqrt(dx * dx + dy * dy + dz * dz);
        }
    }

    // Span and Memory for zero-copy operations
    public static int FindPattern(ReadOnlySpan<byte> data, ReadOnlySpan<byte> pattern)
    {
        for (int i = 0; i <= data.Length - pattern.Length; i++)
        {
            if (data.Slice(i, pattern.Length).SequenceEqual(pattern))
                return i;
        }
        return -1;
    }

    // ValueTask for performance
    public static async ValueTask<int> ComputeAsync(int input)
    {
        if (input < 1000)
        {
            return input * 2; // Synchronous completion
        }

        await Task.Delay(100); // Async operation
        return input * 2;
    }

    // Interlocked operations
    public class ThreadSafeCounter
    {
        private long count = 0;

        public void Increment()
        {
            Interlocked.Increment(ref count);
        }

        public long GetCount()
        {
            return Interlocked.Read(ref count);
        }

        public void Add(long value)
        {
            Interlocked.Add(ref count, value);
        }
    }

    // Memory-mapped files
    public static void ProcessLargeFile(string filePath)
    {
        using var mmf = System.IO.MemoryMappedFiles.MemoryMappedFile.CreateFromFile(filePath);
        using var accessor = mmf.CreateViewAccessor();
        
        var length = accessor.Capacity;
        var buffer = new byte[4096];
        
        for (long offset = 0; offset < length; offset += buffer.Length)
        {
            var bytesToRead = Math.Min(buffer.Length, length - offset);
            accessor.ReadArray(offset, buffer, 0, (int)bytesToRead);
            ProcessBuffer(buffer, (int)bytesToRead);
        }
    }
}

using System;

    }

    // Switch expressions
    public static string GetStatus(int code) => code switch
    {
        200 => "OK",
        404 => "Not Found",
        500 => "Internal Server Error",
        _ => "Unknown"
    };

    // Property patterns
    public static bool IsAdult(Person person) => person switch
    {
        { Age: >= 18 } => true,
        _ => false
    };

    // Tuple patterns
    public static string GetDirection(int x, int y) => (x, y) switch
    {
        (0, 0) => "Origin",
        (_, 0) => "X-axis",
        (0, _) => "Y-axis",
        (var a, var b) when a > 0 && b > 0 => "First quadrant",
        (var a, var b) when a < 0 && b > 0 => "Second quadrant",
        (var a, var b) when a < 0 && b < 0 => "Third quadrant",
        _ => "Fourth quadrant"
    };

    public class AsyncResourcePool<T>
    {
        private readonly SemaphoreSlim semaphore;
        private readonly ConcurrentQueue<T> resources;

        public AsyncResourcePool(IEnumerable<T> initialResources, int maxConcurrency)
        {
            semaphore = new SemaphoreSlim(maxConcurrency);
            resources = new ConcurrentQueue<T>(initialResources);
        }

        public async Task<T> RentAsync()
        {
            await semaphore.WaitAsync();
            
            if (resources.TryDequeue(out T resource))
            {
                return resource;
            }

            throw new InvalidOperationException("No resources available");
        }

        public void Return(T resource)
        {
            resources.Enqueue(resource);
            semaphore.Release();
        }
    }

    // Producer-consumer pattern
    public class ProducerConsumer<T>
    {
        private readonly BlockingCollection<T> queue;
        private readonly CancellationToken cancellationToken;

        public ProducerConsumer(int capacity, CancellationToken cancellationToken)
        {
            queue = new BlockingCollection<T>(capacity);
            this.cancellationToken = cancellationToken;
        }

        public void Produce(T item)
        {
            queue.Add(item, cancellationToken);
        }

        public async Task ConsumeAsync(Func<T, Task> processor)
        {
            await foreach (var item in queue.GetConsumingEnumerable(cancellationToken))
            {
                await processor(item);
            }
        }

        public void CompleteAdding()
        {
            queue.CompleteAdding();
        }
    }

    // Async lock implementation
    public class AsyncLock
    {
        private readonly SemaphoreSlim semaphore = new SemaphoreSlim(1, 1);
        private readonly Task<IDisposable> releaser;

        public AsyncLock()
        {
            releaser = Task.FromResult<IDisposable>(new Releaser(this));
        }

        public Task<IDisposable> LockAsync()
        {
            var wait = semaphore.WaitAsync();
            return wait.IsCompleted ? releaser : wait.ContinueWith(_, releaser.Result, 
                CancellationToken.None, TaskContinuationOptions.ExecuteSynchronously, TaskScheduler.Default);
        }

        private class Releaser : IDisposable
        {
            private readonly AsyncLock toRelease;
            internal Releaser(AsyncLock toRelease) { this.toRelease = toRelease; }
            public void Dispose() { toRelease.semaphore.Release(); }
        }
    }

    // Parallel processing with cancellation
    public static async Task<List<T>> ProcessParallelAsync<T>(
        IEnumerable<T> items,
        Func<T, CancellationToken, Task<T>> processor,
        int maxDegreeOfParallelism,
        CancellationToken cancellationToken = default)
    {
        var results = new ConcurrentBag<T>();
        var semaphore = new SemaphoreSlim(maxDegreeOfParallelism);

        var tasks = items.Select(async item =>
        {
            await semaphore.WaitAsync(cancellationToken);
            try
            {
                var result = await processor(item, cancellationToken);
                results.Add(result);
            }
            finally
            {
                semaphore.Release();
            }
        });

        await Task.WhenAll(tasks);
        return results.ToList();
    }

    // Thread-safe lazy initialization
    public class ThreadSafeLazy<T>
    {
        private readonly Lazy<T> lazy;

        public ThreadSafeLazy(Func<T> factory)
        {
            lazy = new Lazy<T>(factory, LazyThreadSafetyMode.ExecutionAndPublication);
        }

        public T Value => lazy.Value;
    }

    // Async coordination primitives
    public class AsyncBarrier
    {
        private readonly int participantCount;
        private int remainingParticipants;
        private TaskCompletionSource<object> tcs;

        public AsyncBarrier(int participantCount)
        {
            this.participantCount = participantCount;
            this.remainingParticipants = participantCount;
            this.tcs = new TaskCompletionSource<object>();
        }

        public async Task SignalAndWaitAsync()
        {
            var currentTcs = tcs;
            if (Interlocked.Decrement(ref remainingParticipants) == 0)
            {
                currentTcs.SetResult(null);
                remainingParticipants = participantCount;
                tcs = new TaskCompletionSource<object>();
            }
            else
            {
                await currentTcs.Task;
            }
        }
    }
