Language Overview

PHP (Hypertext Preprocessor) is a widely-used open source general-purpose scripting language that is especially suited for web development and can be embedded into HTML. It's known for its simplicity, flexibility, and extensive library support.

Error Handling & Exceptions

<?php
// Custom exception class
class ValidationException extends Exception {
    private array $errors;
    
    public function __construct(string $message, array $errors = []) {
        parent::__construct($message);
        $this->errors = $errors;
    }
    
    public function getErrors(): array {
        return $this->errors;
    }
}

// Try-catch with multiple catch blocks
try {
    $user = new User("", "invalid-email");
    
    if (empty($user->getName())) {
        throw new ValidationException("Name is required", ["name" => "Name cannot be empty"]);
    }
    
    if (!filter_var($user->getEmail(), FILTER_VALIDATE_EMAIL)) {
        throw new ValidationException("Invalid email format", ["email" => "Invalid email address"]);
    }
    
} catch (ValidationException $e) {
    echo "Validation Error: " . $e->getMessage();
    print_r($e->getErrors());
} catch (Exception $e) {
    echo "General Error: " . $e->getMessage();
} finally {
    echo "Cleanup operations...";
}

Database Operations & PDO

<?php
// PDO Database connection
try {
    $pdo = new PDO(
        "mysql:host=localhost;dbname=myapp;charset=utf8mb4",
        "username",
        "password",
        [
            PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
            PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
            PDO::ATTR_EMULATE_PREPARES => false
        ]
    );
} catch (PDOException $e) {
    die("Connection failed: " . $e->getMessage());
}

// Prepared statements
$stmt = $pdo->prepare("SELECT * FROM users WHERE age > ? AND city = ?");
$stmt->execute([18, "NYC"]);
$users = $stmt->fetchAll();

// Insert with prepared statement
$insertStmt = $pdo->prepare("INSERT INTO users (name, email, age) VALUES (?, ?, ?)");
$insertStmt->execute(["Jane", "jane@example.com", 25]);

// Transaction example
try {
    $pdo->beginTransaction();
    
    $stmt1 = $pdo->prepare("UPDATE accounts SET balance = balance - ? WHERE id = ?");
    $stmt1->execute([100, 1]);
    
    $stmt2 = $pdo->prepare("UPDATE accounts SET balance = balance + ? WHERE id = ?");
    $stmt2->execute([100, 2]);
    
    $pdo->commit();
} catch (Exception $e) {
    $pdo->rollBack();
    throw $e;
}

Modern PHP Features & Best Practices

<?php
// Namespaces and autoloading
namespace App\Controllers;

use App\Models\User;
use App\Services\UserService;
use App\Exceptions\ValidationException;

// Trait for common functionality
trait Loggable {
    protected function log(string $message): void {
        error_log("[" . date('Y-m-d H:i:s') . "] {$message}");
    }
}

// Interface for contracts
interface UserRepositoryInterface {
    public function findById(int $id): ?User;
    public function save(User $user): bool;
    public function delete(int $id): bool;
}

// Service class using dependency injection
class UserController {
    private UserService $userService;
    private LoggerInterface $logger;
    
    public function __construct(
        UserService $userService,
        LoggerInterface $logger
    ) {
        $this->userService = $userService;
        $this->logger = $logger;
    }
    
    public function createUser(array $data): JsonResponse {
        try {
            $user = $this->userService->createUser($data);
            $this->logger->info("User created: " . $user->getId());
            
            return new JsonResponse($user, 201);
        } catch (ValidationException $e) {
            $this->logger->error("Validation error: " . $e->getMessage());
            return new JsonResponse(["errors" => $e->getErrors()], 400);
        }
    }
}
