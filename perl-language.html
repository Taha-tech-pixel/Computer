Modern Perl OOP & Moose

#!/usr/bin/perl
use Moose;
use MooseX::StrictConstructor;

# Modern Perl class with Moose
package Person {
    use Moose;
    
    has 'first_name' => (is => 'rw', isa => 'Str', required => 1);
    has 'last_name' => (is => 'rw', isa => 'Str', required => 1);
    has 'age' => (
        is => 'rw',
        isa => 'Int',
        trigger => sub {
            my ($self, $new_age) = @_;
            die "Age cannot be negative" if $new_age < 0;
        }
    );
    
    has 'email' => (
        is => 'rw',
        isa => 'Str',
        lazy => 1,
        builder => '_build_email',
    );
    
    sub _build_email {
        my $self = shift;
        return lc($self->first_name . '.' . $self->last_name . '@example.com');
    }
    
    sub full_name {
        my $self = shift;
        return $self->first_name . ' ' . $self->last_name;
    }
    
    __PACKAGE__->meta->make_immutable;
}

# Role-based composition
package Walkable {
    use Moose::Role;
    requires 'name';
    
    sub walk { $_[0]->name . " is walking" }
}

# Usage
my $person = Person->new(first_name => 'John', last_name => 'Doe', age => 30);
print $person->full_name() . " - " . $person->email() . "\n";

sub map_transform {
    my ($array_ref, $transform_sub) = @_;
    return [map { $transform_sub->($_) } @$array_ref];
}

my @numbers = (1, 2, 3, 4, 5);
my $squares = map_transform(\@numbers, sub { $_[0] ** 2 });
print "Squares: " . join(', ', @$squares) . "\n";

sub transpose_matrix {
    my $matrix_ref = shift;
    my @transposed = ();
    for my $col (0..$#{$matrix_ref->[0]}) {
        for my $row (0..$#{$matrix_ref}) {
            $transposed[$col][$row] = $matrix_ref->[$row][$col];
        }
    }
    return \@transposed;
}

File Processing & System Integration

#!/usr/bin/perl
use strict;
use warnings;
use File::Find;
use DBI;
use JSON;

# Advanced file processing
sub process_log_files {
    my ($directory, $pattern) = @_;
    my %stats = ();
    
    find(sub {
        return unless -f $_ && /$pattern/;
        
        open my $fh, '<', $_ or die "Cannot open $_: $!";
        while (my $line = <$fh>) {
            if ($line =~ /^(\d{4}-\d{2}-\d{2})\s+.*\[(\w+)\]/) {
                $stats{$1}{$2}++;
            }
        }
        close $fh;
    }, $directory);
    
    return \%stats;
}

# Database integration
sub create_db_connection {
    my ($dsn, $user, $password) = @_;
    return DBI->connect($dsn, $user, $password, {
        RaiseError => 1,
        AutoCommit => 0,
    });
}

# System monitoring
sub monitor_system {
    my $stats = {};
    
    # Memory usage from /proc/meminfo
    if (open my $meminfo, '<', '/proc/meminfo') {
        my %mem = ();
        while (my $line = <$meminfo>) {
            if ($line =~ /^(\w+):\s+(\d+)\s+kB$/) {
                $mem{$1} = $2;
            }
        }
        $stats->{memory_usage} = sprintf("%.2f", 
            (($mem{MemTotal} - $mem{MemAvailable}) / $mem{MemTotal}) * 100);
        close $meminfo;
    }
    
    return $stats;
}

# Configuration file parsing
sub parse_config {
    my ($filename) = @_;
    my %config = ();
    my $section = 'default';
    
    open my $fh, '<', $filename or die "Cannot open $filename: $!";
    while (my $line = <$fh>) {
        chomp $line;
        next if $line =~ /^#/ || $line eq '';
        
        if ($line =~ /^\[(.+)\]$/) {
            $section = $1;
        } elsif ($line =~ /^(.+?)\s*=\s*(.+)$/) {
            $config{$section}{$1} = $2;
        }
    }
    close $fh;
    return \%config;
}

# Postmodern object system
use Object::Pad;

class BankAccount {
    has $balance :param = 0;
    has $owner :param :reader;
    
    method deposit($amount) {
        die "Amount must be positive" if $amount <= 0;
        $balance += $amount;
        return $balance;
    }
    
    method withdraw($amount) {
        die "Insufficient funds" if $amount > $balance;
        $balance -= $amount;
        return $balance;
    }
    
    method balance() { return $balance }
}

$account->deposit(500);

# Performance optimization with memoization
sub fibonacci {
    my $n = shift;
    return $n if $n <= 1;
    return fibonacci($n - 1) + fibonacci($n - 2);
}
memoize('fibonacci');

# Memory-efficient processing
sub process_large_file {
    my ($filename) = @_;
    my %word_count = ();
    
    open my $fh, '<', $filename or die $!;
    while (my $line = <$fh>) {
        my @words = split /\W+/, lc($line);
        $word_count{$_}++ for grep { length > 2 } @words;
    }
    close $fh;
    
    # Return top 10 words
    my @top = sort { $word_count{$b} <=> $word_count{$a} } keys %word_count;
    return [map { { word => $_, count => $word_count{$_} } } splice(@top, 0, 10)];
}
