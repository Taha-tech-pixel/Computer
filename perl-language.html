Basic Syntax

Hello World

#!/usr/bin/perl
use strict;
use warnings;

print "Hello, World!\n";

Variables

my $scalar = "Hello";           # Scalar
my @array = (1, 2, 3);         # Array
my %hash = (key => "value");    # Hash

print "$scalar\n";
print "@array\n";
print "$hash{key}\n";

Regular Expressions

my $text = "Hello World";
if ($text =~ /World/) {
    print "Found 'World'\n";
}

$text =~ s/World/Perl/;
print "$text\n";  # Hello Perl

Conditionals

Loops

File Operations

Reading Files

open(my $fh, '<', 'file.txt') or die "Cannot open file: $!";
while (my $line = <$fh>) {
    chomp $line;
    print "$line\n";
}
close $fh;

Writing Files

open(my $fh, '>', 'output.txt') or die "Cannot create file: $!";
print $fh "Hello World\n";
close $fh;

Subroutines

Basic Subroutine

sub greet {
    my ($name) = @_;
    return "Hello, $name!";
}

my $message = greet("World");
print "$message\n";

Subroutine with Default Values

sub multiply {
    my ($a, $b) = @_;
    $b //= 1;  # Default value
    return $a * $b;
}

print multiply(5);      # 5
print multiply(5, 3);   # 15

Modules and Packages

Using CPAN Modules

use strict;

my $content = get("http://example.com");
my $data = decode_json($content);

Creating Packages

package MyModule;

sub new {
    my $class = shift;
    my $self = {};
    bless $self, $class;
    return $self;
}

sub method {
    my $self = shift;
    return "Hello from method";
}

1;  # Return true

Basic Error Handling

    # Risky code

Using Carp

use Carp;

sub process_data {
    my ($data) = @_;

    }

}
